Messages are stored in Supabase table public.messages, synced in realtime, and filtered by room.

Run this SQL in Supabase (create or upgrade) 
create table if not exists public.messages (
  id bigint generated by default as identity primary key,
  room_id text not null default 'general' check (char_length(room_id) between 1 and 32),
  username text not null check (char_length(username) between 1 and 24),
  content text not null check (char_length(content) between 1 and 500),
  created_at timestamptz not null default now()
);

do $$
begin

    where conname = 'messages_content_length'
  ) then
    alter table public.messages

    check (char_length(content) between 1 and 500);
  end if;

  if not exists (

    from pg_constraint

  ) then
